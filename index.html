  if(!pend[idx]) return;

  var q = Math.max(1, parseFloat(val) || 0);

  pend[idx].qty = q;

  

  // Ricalcolo proporzionale dei macro rispetto alla porzione estratta dall'IA

  var ratio = q / pend[idx].base_qty;

  pend[idx].kcal = Math.round(pend[idx].base_kcal * ratio);

  pend[idx].prot = Math.round(pend[idx].base_prot * ratio * 10) / 10;

  

  // Ottimizzazione di aggiornamento parziale per evitare sfarfallio

  var items = document.querySelectorAll('.rev-item');

  if (items[idx]) {

    var midRow = items[idx].querySelector('.rev-row-mid');

    midRow.innerHTML = '<span>Calorie: <strong style="color:var(--blue)">'+Math.round(pend[idx].kcal)+' kcal</strong></span> &middot; '

                     + '<span>Proteine: <strong style="color:var(--blue2)">'+Math.round(pend[idx].prot)+'g</strong></span>';

    

    var numInp = items[idx].querySelector('.rev-qty-val');

    if (numInp && numInp.value != q) numInp.value = q;

    

    var rangeInp = items[idx].querySelector('.rev-slider');

    if (rangeInp && rangeInp.value != q) rangeInp.value = q;

  }

  

  updateReviewTotals();

}
